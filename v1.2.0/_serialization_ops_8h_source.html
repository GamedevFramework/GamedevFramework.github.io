373 template<typename T>
374 inline
375 Deserializer& operator|(Deserializer& ar, Span<T> array) {
376 std::size_t size;
377
378 if (!ar.readSizeHeader(size)) {
379 return ar;
380 }
381
382 if (size != array.getSize()) {
383 return ar;
384 }
385
386 for (auto& item : array) {
387 ar | item;
388 }
389
390 return ar;
391 }
