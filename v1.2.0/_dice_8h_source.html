68 constexpr Dice(std::string_view str)
69 : m_faces(0)
70 , m_count(0)
71 , m_modifier(0)
72 {
73 std::size_t i = 0;
74 std::size_t sz = str.size();
75
76 while (i < sz && '0' <= str[i] && str[i] <= '9') {
77 m_count = m_count * 10 + (str[i] - '0');
78 ++i;
79 }
80
81 if (i == sz) {
82 return;
83 }
84
85 if (str[i] != 'd' && str[i] != 'D') {
86 return;
87 }
88
89 if (m_count == 0) {
90 m_count = 1;
91 }
92
93 ++i; // eat 'd'
94
95 while (i < sz && '0' <= str[i] && str[i] <= '9') {
96 m_faces = m_faces * 10 + (str[i] - '0');
97 ++i;
98 }
99
100 if (i == sz) {
101 return;
102 }
103
104 if (str[i] != '+') {
105 return;
106 }
107
108 ++i; // eat '+'
109
110 while (i < sz && '0' <= str[i] && str[i] <= '9') {
111 m_modifier = m_modifier * 10 + (str[i] - '0');
112 ++i;
113 }
114 }
