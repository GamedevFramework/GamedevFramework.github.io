352 template<typename T>
353 inline
354 Deserializer& operator|(Deserializer& ar, Span<T> array) {
355 std::size_t size;
356
357 if (!ar.readSizeHeader(size)) {
358 return ar;
359 }
360
361 if (size != array.getSize()) {
362 return ar;
363 }
364
365 for (auto& item : array) {
366 ar | item;
367 }
368
369 return ar;
370 }
