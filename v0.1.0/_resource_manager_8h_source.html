66  /**
67  * @brief Get a ressource
68  *
69  * If the ressource exists in the cache, it is returned immediately.
70  * Otherwise, it is searched thanks to an asset manager and put in the
71  * cache.
72  *
73  * @param assetManager An asset manager
74  * @param filename The filename of the ressource
75  * @return The ressource or `nullptr` if it has not been found
76  */
77  T *getResource(AssetManager& assetManager, const Path& filename) {
78  auto it = m_cache.find(filename);
79 
80  if (it != m_cache.end()) {
81  return it->second.get();
82  }
83 
84  Path absolutePath = assetManager.getAbsolutePath(filename);
85 
86  if (absolutePath.empty()) {
87  return nullptr;
88  }
89 
90  auto ptr = m_loader(absolutePath);
91 
92  if (!ptr) {
93  return nullptr;
94  }
95 
96  auto inserted = m_cache.emplace(filename, std::move(ptr));
97 
98  if (inserted.second) {
99  return inserted.first->second.get();
100  }
101 
102  return nullptr;
103  }
104 
105  private:
106  Loader m_loader;
107  std::map<Path, std::unique_ptr<T>> m_cache;
108  };
